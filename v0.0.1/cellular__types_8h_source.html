488typedef struct CellularPsmSettings
489{
490 /*
491 * 0 = PSM Disable
492 * 1 = PSM Enable.
493 */
494 uint8_t mode;
496 /*
497 * Bits 5 to 1 represent the binary coded timer value
498 * Bits 6 to 8 define the timer value unit as follows:
499 * Bits
500 * 8 7 6
501 * 0 0 0 value is incremented in multiples of 10 minutes
502 * 0 0 1 value is incremented in multiples of 1 hour
503 * 0 1 0 value is incremented in multiples of 10 hours
504 * 0 1 1 value is incremented in multiples of 2 seconds
505 * 1 0 0 value is incremented in multiples of 30 seconds
506 * 1 0 1 value is incremented in multiples of 1 minute
507 *
508 * e.g. "00001010" equals to 100 minutes.
509 * first uint8_t is used for PSM set, whole uint32_t is used for PSM get.
510 */
511 uint32_t periodicTauValue;
513 /*
514 * Bits 5 to 1 represent the binary coded timer value
515 * Bits 6 to 8 define the timer value unit as follows:
516 * Bits
517 * 8 7 6
518 * 0 0 0 value is incremented in multiples of 10 minutes
519 * 0 0 1 value is incremented in multiples of 1 hour
520 * 0 1 0 value is incremented in multiples of 10 hours
521 * 0 1 1 value is incremented in multiples of 2 seconds
522 * 1 0 0 value is incremented in multiples of 30 seconds
523 * 1 0 1 value is incremented in multiples of 1 minute
524 *
525 * e.g. "00001010" equals to 100 minutes.
526 * first uint8_t is used for PSM set, whole uint32_t is used for PSM get.
527 */
528 uint32_t periodicRauValue;
530 /*
531 * Bits 5 to 1 represent the binary coded timer value
532 * Bits 6 to 8 define the timer value unit as follows:
533 * Bits
534 * 8 7 6
535 * 0 0 0 value is incremented in multiples of 10 minutes
536 * 0 0 1 value is incremented in multiples of 1 hour
537 * 0 1 0 value is incremented in multiples of 10 hours
538 * 0 1 1 value is incremented in multiples of 2 seconds
539 * 1 0 0 value is incremented in multiples of 30 seconds
540 * 1 0 1 value is incremented in multiples of 1 minute
541 *
542 * e.g. "00001010" equals to 100 minutes.
543 * first uint8_t is used for PSM set, whole uint32_t is used for PSM get.
544 */
545 uint32_t gprsReadyTimer;
547 /*
548 * Bits 5 to 1 represent the binary coded timer value.
549 * Bits 6 to 8 define the timer value unit as follows:
550 * Bits
551 * 8 7 6
552 * 0 0 0 value is incremented in multiples of 2 seconds
553 * 0 0 1 value is incremented in multiples of 1 minute
554 * 0 1 0 value is incremented in multiples of decihours
555 * 1 1 1 value indicates that the timer is deactivated.
556 *
557 * "00001111" equals to 30 seconds.
558 * first uint8_t is used for PSM set, whole uint32_t is used for PSM get.
559 */
560 uint32_t activeTimeValue;
561} CellularPsmSettings_t;

567typedef struct CellularEidrxSettings
568{
569 /*
570 * 0 Disable the use of e-I-DRX
571 * 1 Enable the use of e-I-DRX
572 * 2 Enable the use of e-I-DRX and enable the unsolicited result code
573 * 3 Disable the use of e-I-DRX and discard all parameters for e-I-DRX or,
574 * if available, reset to the manufacturer specific default values.
575 */
576 uint8_t mode;
578 /*
579 * 2 GSM
580 * 3 UTRAN
581 * 4 LTE Cat M1
582 * 5 LTE Cat NB1
583 */
584 uint8_t rat;
586 /*
587 * String type. Half a byte in a 4 bit format.
588 * bit
589 * 4 3 2 1 E-UTRAN e-I-DRX cycle length duration
590 * 0 0 0 0 5.12 seconds
591 * 0 0 0 1 10.24 seconds
592 * 0 0 1 0 20.48 seconds
593 * 0 0 1 1 40.96 seconds
594 * 0 1 0 0 61.44 seconds
595 * 0 1 0 1 81.92 seconds
596 * 0 1 1 0 102.4 seconds
597 * 0 1 1 1 122.88 seconds
598 * 1 0 0 0 143.36 seconds
599 * 1 0 0 1 163.84 seconds
600 * 1 0 1 0 327.68 seconds
601 * 1 0 1 1 655,36 seconds
602 * 1 1 0 0 1310.72 seconds
603 * 1 1 0 1 2621.44 seconds
604 * 1 1 1 0 5242.88 seconds
605 * 1 1 1 1 10485.76 seconds
606 */
607 uint8_t requestedEdrxVaue;
608 uint8_t nwProvidedEdrxVaue;
610 /*
611 * LTE Cat M1 mode
612 * bit
613 * 4 3 2 1 Paging Time Window length
614 * 0 0 0 0 1.28 seconds
615 * 0 0 0 1 2.56 seconds
616 * 0 0 1 0 3.84 seconds
617 * 0 0 1 1 5.12 seconds
618 * 0 1 0 0 6.4 seconds
619 * 0 1 0 1 7.68 seconds
620 * 0 1 1 0 8.96 seconds
621 * 0 1 1 1 10.24 seconds
622 * 1 0 0 0 11.52 seconds
623 * 1 0 0 1 12.8 seconds
624 * 1 0 1 0 14.08 seconds
625 * 1 0 1 1 15.36 seconds
626 * 1 1 0 0 16.64 seconds
627 * 1 1 0 1 17.92 seconds
628 * 1 1 1 0 19.20 seconds
629 * 1 1 1 1 20.48 seconds
630 *
631 * LTE Cat NB1 mode
632 * bit
633 * 4 3 2 1 Paging Time Window length
634 * 0 0 0 0 2.56 seconds
635 * 0 0 0 1 5.12 seconds
636 * 0 0 1 0 7.68 seconds
637 * 0 0 1 1 10.24 seconds
638 * 0 1 0 0 12.8 seconds
639 * 0 1 0 1 15.36 seconds
640 * 0 1 1 0 17.92 seconds
641 * 0 1 1 1 20.48 seconds
642 * 1 0 0 0 23.04 seconds
643 * 1 0 0 1 25.6 seconds
644 * 1 0 1 0 28.16 seconds
645 * 1 0 1 1 30.72 seconds
646 * 1 1 0 0 33.28 seconds
647 * 1 1 0 1 35.84 seconds
648 * 1 1 1 0 38.4 seconds
649 * 1 1 1 1 40.96 seconds
650 */
651 uint8_t pagingTimeWindow;
